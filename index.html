import pandas as pd
import matplotlib.pyplot as plt

# Load data
df = pd.read_csv(r'plant_growth_data.csv')

# Calculate success rates
soil_success = df.groupby('Soil_Type')['Growth_Milestone'].mean()
fert_success = df.groupby('Fertilizer_Type')['Growth_Milestone'].mean()
water_success = df.groupby('Water_Frequency')['Growth_Milestone'].mean()

# Plot
fig, axes = plt.subplots(1, 3, figsize=(18, 5))
soil_success.plot(kind='bar', ax=axes[0], title='Soil Type Impact', ylabel='Success Rate')
fert_success.plot(kind='bar', ax=axes[1], title='Fertilizer Impact', color='green')
water_success.plot(kind='bar', ax=axes[2], title='Water Frequency Impact', color='orange')
plt.tight_layout()
plt.savefig("Water Frequency Impact.png")
plt.show()
